<script setup>
import { ref, computed } from 'vue';
import ModCard from '../components/ui/ModCard.vue';
import { mods } from '../data/mods.js';

const categories = ['All', 'Game Mod', 'App Mod', 'Script'];
const activeFilter = ref('All');
const viewMode = ref('grid'); // 'grid' or 'list'

const filteredMods = computed(() => {
  if (activeFilter.value === 'All') {
    return mods;
  }
  return mods.filter(mod => mod.category === activeFilter.value);
});

const formattedDate = (dateString) => {
  if (!dateString) return '';
  return new Date(dateString).toLocaleDateString('en-US', {
    year: 'numeric', month: 'short', day: 'numeric'
  });
};
</script>

<template>
  <div class="page-container fade-in-up">
    <div class="explorer-window" v-animate-on-scroll>
      <div class="window-header">
        <div class="header-title">/home/jiya/mods</div>
        <div class="view-toggle">
          <button @click="viewMode = 'list'" :class="{ active: viewMode === 'list' }" aria-label="List View">
            <v-icon name="fa-list" />
          </button>
          <button @click="viewMode = 'grid'" :class="{ active: viewMode === 'grid' }" aria-label="Grid View">
            <v-icon name="fa-th-large" />
          </button>
        </div>
      </div>

      <div class="window-toolbar">
        <div class="filter-tabs">
          <button 
            v-for="category in categories" 
            :key="category"
            :class="{ active: activeFilter === category }"
            @click="activeFilter = category"
          >
            <span>{{ category }}</span>
          </button>
        </div>
      </div>
      
      <div class="window-content">
        <TransitionGroup v-if="viewMode === 'grid'" name="list" tag="div" class="mods-grid">
          <ModCard 
            v-for="(mod) in filteredMods" 
            :key="mod.id"
            :mod="mod"
          />
        </TransitionGroup>

        <TransitionGroup v-if="viewMode === 'list'" name="list" tag="div" class="mods-list">
          <!-- Header Daftar -->
          <div class="list-header">
            <span class="col-name">Name</span>
            <span class="col-version">Version</span>
            <span class="col-updated">Last Updated</span>
          </div>
          <!-- Item Daftar -->
          <router-link
            v-for="mod in filteredMods"
            :key="mod.id"
            :to="`/mods/${mod.slug}`"
            class="list-item"
          >
            <div class="col-name">
              <v-icon :name="mod.category === 'Game Mod' ? 'io-game-controller' : (mod.category === 'App Mod' ? 'fa-mobile-alt' : 'fa-code')" />
              <span>{{ mod.title }}</span>
            </div>
            <span class="col-version">{{ mod.version }}</span>
            <span class="col-updated">{{ formattedDate(mod.lastUpdated) }}</span>
          </router-link>
        </TransitionGroup>
      </div>
    </div>
  </div>
</template>
